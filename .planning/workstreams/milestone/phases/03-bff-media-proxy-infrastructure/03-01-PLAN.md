<!-- Reference implementation: existing backendFetch() that imageBackendFetch() mirrors -->

From frontend/admin/src/lib/api-client.js:
```javascript
const BACKEND_URL = process.env.BACKEND_URL;

export async function backendFetch(path, options = {}) {
  const { headers = {}, ...rest } = options;

  const res = await fetch(`${BACKEND_URL}${path}`, {
    ...rest,
    headers: {
      'Content-Type': 'application/json',
      ...headers,
    },
  });

  const data = await res.json().catch(() => null);

  return { ok: res.ok, status: res.status, data };
}
```

<!-- image_backend auth mechanism (what X-API-Key header must match) -->

From image_backend/src/api/dependencies/auth.py:
```python
async def verify_api_key(
    x_api_key: str | None = Header(None, alias="X-API-Key"),
    api_key: str | None = Query(None),
) -> None:
    key = x_api_key or api_key
    internal_key = settings.INTERNAL_API_KEY.get_secret_value()
    if not internal_key:
        return  # auth disabled in dev
    if not key or not hmac.compare_digest(key, internal_key):
        raise UnauthorizedError(
            message="Invalid API key.",
            error_code="INVALID_API_KEY",
        )
```
