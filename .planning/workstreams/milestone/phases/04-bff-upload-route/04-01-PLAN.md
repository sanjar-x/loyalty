export async function POST(request) {
  const token = await getAccessToken();
  if (!token) {
    return NextResponse.json(
      { error: { code: 'UNAUTHORIZED', message: 'Not authenticated', details: {} } },
      { status: 401 },
    );
  }
  const body = await request.json();
  const { ok, status, data } = await backendFetch('/api/v1/catalog/products', {
    method: 'POST',
    headers: { Authorization: `Bearer ${token}`, 'Content-Type': 'application/json' },
    body: JSON.stringify(body),
  });
  if (!ok) {
    return NextResponse.json(
      data ?? { error: { code: 'SERVICE_UNAVAILABLE', message: 'Backend unavailable', details: {} } },
      { status: status || 502 },
    );
  }
  return NextResponse.json(data, { status: 201 });
}

```javascript
import { NextResponse } from 'next/server';
import { imageBackendFetch } from '@/lib/image-api-client';
import { getAccessToken } from '@/lib/auth';

export async function POST(request) {
  const token = await getAccessToken();
  if (!token) {
    return NextResponse.json(
      { error: { code: 'UNAUTHORIZED', message: 'Not authenticated', details: {} } },
      { status: 401 },
    );
  }

  const body = await request.json();

  // Strip product-specific fields that image_backend does not accept.
  // image_backend UploadRequest only accepts: { contentType: string, filename?: string }
  // Frontend may send: { mediaType, role, contentType, sortOrder, filename }
  const { contentType, filename } = body;
  const imagePayload = { contentType, filename };

  const { ok, status, data } = await imageBackendFetch('/api/v1/media/upload', {
    method: 'POST',
    body: JSON.stringify(imagePayload),
  });

  if (!ok) {
    return NextResponse.json(
      data ?? { error: { code: 'IMAGE_SERVICE_ERROR', message: 'Image service error', details: {} } },
      { status: status || 502 },
    );
  }

  return NextResponse.json(data, { status: 201 });
}
```

Key implementation notes:
- Uses `imageBackendFetch` (NOT `backendFetch`) — routes to image_backend, not main backend
- Auth check via `getAccessToken()` prevents unauthenticated proxy usage. Note: the JWT token is NOT forwarded to image_backend. `imageBackendFetch()` handles service auth via X-API-Key internally.
- Destructures only `contentType` and `filename` from body, discarding `mediaType`, `role`, `sortOrder` — image_backend would 422 on extra fields (Pydantic strict validation)
- Returns 201 on success (matching image_backend's 201 response)
- Error response uses standard envelope `{ error: { code, message, details } }` matching codebase convention
- Does NOT modify image_backend response — passes `storageObjectId`, `presignedUrl`, `expiresIn` through as-is
- Does NOT delete the existing broken route at `src/app/api/catalog/products/[productId]/media/upload/route.js` — Phase 6 will redirect frontend to use this new route
